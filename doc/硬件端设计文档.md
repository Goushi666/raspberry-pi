#### 3.1.1 DHT22温湿度传感器 (sensors/dht22.py)
```python
import Adafruit_DHT
import time

class DHT22Sensor:
    def __init__(self, pin):
        self.sensor = Adafruit_DHT.DHT22
        self.pin = pin

    def read(self):
        """读取温湿度数据"""
        humidity, temperature = Adafruit_DHT.read_retry(self.sensor, self.pin)

        if humidity is not None and temperature is not None:
            return {
                'temperature': round(temperature, 1),
                'humidity': round(humidity, 1),
                'timestamp': int(time.time() * 1000)
            }
        return None

    def is_valid(self, data):
        """数据有效性检查"""
        if data is None:
            return False
        return (-40 <= data['temperature'] <= 80 and
                0 <= data['humidity'] <= 100)
```

#### 3.1.2 BH1750光照传感器 (sensors/bh1750.py)
```python
import smbus2
import time

class BH1750Sensor:
    ADDR = 0x23
    ONE_TIME_HIGH_RES = 0x20

    def __init__(self, bus=1):
        self.bus = smbus2.SMBus(bus)

    def read(self):
        """读取光照强度"""
        self.bus.write_byte(self.ADDR, self.ONE_TIME_HIGH_RES)
        time.sleep(0.2)

        data = self.bus.read_i2c_block_data(self.ADDR, 0x00, 2)
        light = (data[0] << 8 | data[1]) / 1.2

        return {
            'light': round(light, 1),
            'timestamp': int(time.time() * 1000)
        }

#### 3.1.3 数据采集管理器 (sensors/manager.py)
```python
import threading
import time
from collections import deque

class SensorManager:
    def __init__(self, dht22, bh1750, interval=5):
        self.dht22 = dht22
        self.bh1750 = bh1750
        self.interval = interval
        self.running = False
        self.data_queue = deque(maxlen=100)
        self.thread = None

    def start(self):
        """启动数据采集"""
        self.running = True
        self.thread = threading.Thread(target=self._collect_loop)
        self.thread.start()

    def _collect_loop(self):
        """采集循环"""
        while self.running:
            try:
                # 读取传感器数据
                temp_hum = self.dht22.read()
                light = self.bh1750.read()

                if temp_hum and light:
                    data = {**temp_hum, **light}
                    # 数据滤波
                    filtered = self._filter(data)
                    self.data_queue.append(filtered)

            except Exception as e:
                print(f"采集错误: {e}")

            time.sleep(self.interval)

    def _filter(self, data):
        """移动平均滤波"""
        if len(self.data_queue) < 3:
            return data

        recent = list(self.data_queue)[-3:]
        return {
            'temperature': sum(d['temperature'] for d in recent) / 3,
            'humidity': sum(d['humidity'] for d in recent) / 3,
            'light': sum(d['light'] for d in recent) / 3,
            'timestamp': data['timestamp']
        }

    def get_latest(self):
        """获取最新数据"""
        return self.data_queue[-1] if self.data_queue else None

    def stop(self):
        """停止采集"""
        self.running = False
        if self.thread:
            self.thread.join()
```

#### 3.2.1 L298N驱动 (motor/driver.py)
```python
import RPi.GPIO as GPIO

class L298NDriver:
    def __init__(self, ena, in1, in2, enb, in3, in4):
        self.ena = ena
        self.in1 = in1
        self.in2 = in2
        self.enb = enb
        self.in3 = in3
        self.in4 = in4

        GPIO.setmode(GPIO.BCM)
        GPIO.setup([ena, in1, in2, enb, in3, in4], GPIO.OUT)

        # PWM初始化
        self.pwm_left = GPIO.PWM(ena, 1000)
        self.pwm_right = GPIO.PWM(enb, 1000)
        self.pwm_left.start(0)
        self.pwm_right.start(0)

    def set_left_motor(self, speed):
        """设置左电机速度 (-100 ~ 100)"""
        if speed > 0:
            GPIO.output(self.in1, GPIO.HIGH)
            GPIO.output(self.in2, GPIO.LOW)
        elif speed < 0:
            GPIO.output(self.in1, GPIO.LOW)
            GPIO.output(self.in2, GPIO.HIGH)
        else:
            GPIO.output(self.in1, GPIO.LOW)
            GPIO.output(self.in2, GPIO.LOW)

        self.pwm_left.ChangeDutyCycle(abs(speed))

    def set_right_motor(self, speed):
        """设置右电机速度 (-100 ~ 100)"""
        if speed > 0:
            GPIO.output(self.in3, GPIO.HIGH)
            GPIO.output(self.in4, GPIO.LOW)
        elif speed < 0:
            GPIO.output(self.in3, GPIO.LOW)
            GPIO.output(self.in4, GPIO.HIGH)
        else:
            GPIO.output(self.in3, GPIO.LOW)
            GPIO.output(self.in4, GPIO.LOW)

        self.pwm_right.ChangeDutyCycle(abs(speed))

    def stop(self):
        """停止所有电机"""
        self.set_left_motor(0)
        self.set_right_motor(0)

    def cleanup(self):
        """清理GPIO"""
        self.stop()
        self.pwm_left.stop()
        self.pwm_right.stop()
        GPIO.cleanup()
```

#### 3.2.2 运动控制器 (motor/controller.py)
```python
import time

class VehicleController:
    def __init__(self, driver):
        self.driver = driver
        self.mode = 'manual'
        self.last_command_time = 0
        self.timeout = 5  # 5秒超时

    def forward(self, speed=50):
        """前进"""
        self.driver.set_left_motor(speed)
        self.driver.set_right_motor(speed)
        self.last_command_time = time.time()

    def backward(self, speed=50):
        """后退"""
        self.driver.set_left_motor(-speed)
        self.driver.set_right_motor(-speed)
        self.last_command_time = time.time()

    def turn_left(self, speed=50):
        """左转"""
        self.driver.set_left_motor(speed * 0.3)
        self.driver.set_right_motor(speed)
        self.last_command_time = time.time()

    def turn_right(self, speed=50):
        """右转"""
        self.driver.set_left_motor(speed)
        self.driver.set_right_motor(speed * 0.3)
        self.last_command_time = time.time()

    def stop(self):
        """停止"""
        self.driver.stop()

    def check_timeout(self):
        """检查超时"""
        if time.time() - self.last_command_time > self.timeout:
            self.stop()
            return True
        return False
```

#### 3.3.1 摄像头控制 (vision/camera.py)
```python
import cv2
import threading

class Camera:
    def __init__(self, resolution=(640, 480), framerate=30):
        self.cap = cv2.VideoCapture(0)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, resolution[0])
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, resolution[1])
        self.cap.set(cv2.CAP_PROP_FPS, framerate)
        self.frame = None
        self.running = False

    def start(self):
        """启动采集"""
        self.running = True
        threading.Thread(target=self._capture_loop, daemon=True).start()

    def _capture_loop(self):
        """采集循环"""
        while self.running:
            ret, frame = self.cap.read()
            if ret:
                self.frame = frame

    def read(self):
        """读取当前帧"""
        return self.frame

    def stop(self):
        """停止采集"""
        self.running = False
        self.cap.release()
```

#### 3.3.2 图像处理器 (vision/processor.py)
```python
import cv2
import numpy as np

class ImageProcessor:
    def __init__(self, threshold=127):
        self.threshold = threshold

    def preprocess(self, frame):
        """图像预处理"""
        # 灰度化
        gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
        # 高斯模糊
        blurred = cv2.GaussianBlur(gray, (5, 5), 0)
        # 二值化
        _, binary = cv2.threshold(blurred, self.threshold, 255, cv2.THRESH_BINARY)
        return binary

    def find_line_center(self, binary):
        """查找轨迹中心线"""
        height, width = binary.shape
        roi = binary[int(height * 0.6):height, :]

        # 轮廓检测
        contours, _ = cv2.findContours(roi, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if not contours:
            return None

        # 找到最大轮廓
        largest = max(contours, key=cv2.contourArea)
        M = cv2.moments(largest)

        if M['m00'] == 0:
            return None

        # 计算中心点
        cx = int(M['m10'] / M['m00'])
        return cx

#### 3.3.3 循迹控制器 (vision/tracker.py)
```python
from utils.pid import PIDController

class LineTracker:
    def __init__(self, camera, processor, controller):
        self.camera = camera
        self.processor = processor
        self.controller = controller
        self.pid = PIDController(kp=0.5, ki=0.1, kd=0.2)
        self.running = False
        self.base_speed = 40

    def start(self):
        """启动循迹"""
        self.running = True
        self.camera.start()

    def update(self):
        """循迹更新"""
        if not self.running:
            return

        frame = self.camera.read()
        if frame is None:
            return

        # 图像处理
        binary = self.processor.preprocess(frame)
        center_x = self.processor.find_line_center(binary)

        if center_x is None:
            self.controller.stop()
            return

        # 计算偏差
        frame_center = frame.shape[1] // 2
        deviation = center_x - frame_center

        # PID控制
        correction = self.pid.compute(deviation)

        # 调整电机速度
        left_speed = self.base_speed - correction
        right_speed = self.base_speed + correction

        self.controller.driver.set_left_motor(left_speed)
        self.controller.driver.set_right_motor(right_speed)

        return {'deviation': deviation, 'correction': correction}

    def stop(self):
        """停止循迹"""
        self.running = False
        self.controller.stop()
        self.camera.stop()
```

### 3.4 通信模块

#### 3.4.1 MQTT客户端 (communication/mqtt_client.py)
```python
import paho.mqtt.client as mqtt
import json
import time

class MQTTClient:
    def __init__(self, broker, port=1883):
        self.broker = broker
        self.port = port
        self.client = mqtt.Client()
        self.connected = False
        self.callbacks = {}

        self.client.on_connect = self._on_connect
        self.client.on_disconnect = self._on_disconnect
        self.client.on_message = self._on_message

    def connect(self):
        """连接到MQTT Broker"""
        try:
            self.client.connect(self.broker, self.port, 60)
            self.client.loop_start()
        except Exception as e:
            print(f"连接失败: {e}")

    def _on_connect(self, client, userdata, flags, rc):
        """连接回调"""
        if rc == 0:
            self.connected = True
            print("MQTT连接成功")
            # 订阅控制主题
            self.client.subscribe("vehicle/control")
        else:
            print(f"连接失败，返回码: {rc}")

    def _on_disconnect(self, client, userdata, rc):
        """断开回调"""
        self.connected = False
        print("MQTT连接断开")

    def _on_message(self, client, userdata, msg):
        """消息回调"""
        try:
            data = json.loads(msg.payload.decode())
            topic = msg.topic

            if topic in self.callbacks:
                self.callbacks[topic](data)
        except Exception as e:
            print(f"消息处理错误: {e}")

    def subscribe(self, topic, callback):
        """订阅主题"""
        self.callbacks[topic] = callback
        self.client.subscribe(topic)

    def publish(self, topic, data):
        """发布消息"""
        if self.connected:
            payload = json.dumps(data)
            self.client.publish(topic, payload)

    def disconnect(self):
        """断开连接"""
        self.client.loop_stop()
        self.client.disconnect()
```

#### 3.5.1 PID控制器 (utils/pid.py)
```python
import time

class PIDController:
    def __init__(self, kp, ki, kd):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.last_error = 0
        self.integral = 0
        self.last_time = time.time()

    def compute(self, error):
        """计算PID输出"""
        current_time = time.time()
        dt = current_time - self.last_time

        if dt <= 0:
            dt = 0.01

        # 比例项
        p = self.kp * error

        # 积分项
        self.integral += error * dt
        i = self.ki * self.integral

        # 微分项
        derivative = (error - self.last_error) / dt
        d = self.kd * derivative

        # 更新状态
        self.last_error = error
        self.last_time = current_time

        return p + i + d

    def reset(self):
        """重置PID"""
        self.last_error = 0
        self.integral = 0
        self.last_time = time.time()
```

## 4. 主程序设计

### 4.1 主程序流程 (main.py)
```python
import yaml
import signal
import sys
from sensors.dht22 import DHT22Sensor
from sensors.bh1750 import BH1750Sensor
from sensors.manager import SensorManager
from motor.driver import L298NDriver
from motor.controller import VehicleController
from vision.camera import Camera
from vision.processor import ImageProcessor
from vision.tracker import LineTracker
from communication.mqtt_client import MQTTClient

class MainApp:
    def __init__(self, config_path='config/config.yaml'):
        self.config = self.load_config(config_path)
        self.init_components()

    def load_config(self, path):
        """加载配置"""
        with open(path, 'r') as f:
            return yaml.safe_load(f)

    def init_components(self):
        """初始化组件"""
        # 传感器
        self.dht22 = DHT22Sensor(self.config['sensors']['dht22_pin'])
        self.bh1750 = BH1750Sensor()
        self.sensor_manager = SensorManager(self.dht22, self.bh1750)

        # 电机
        motor_cfg = self.config['motor']
        self.driver = L298NDriver(**motor_cfg)
        self.controller = VehicleController(self.driver)

        # 视觉
        self.camera = Camera()
        self.processor = ImageProcessor()
        self.tracker = LineTracker(self.camera, self.processor, self.controller)

        # 通信
        mqtt_cfg = self.config['mqtt']
        self.mqtt = MQTTClient(mqtt_cfg['broker'], mqtt_cfg['port'])
        self.mqtt.subscribe('vehicle/control', self.handle_control)

    def handle_control(self, data):
        """处理控制指令"""
        action = data.get('action')
        speed = data.get('speed', 50)

        if action == 'forward':
            self.controller.forward(speed)
        elif action == 'backward':
            self.controller.backward(speed)
        elif action == 'left':
            self.controller.turn_left(speed)
        elif action == 'right':
            self.controller.turn_right(speed)
        elif action == 'stop':
            self.controller.stop()
        elif action == 'auto':
            self.tracker.start()
        elif action == 'manual':
            self.tracker.stop()

    def run(self):
        """运行主程序"""
        self.sensor_manager.start()
        self.mqtt.connect()

        while True:
            # 上报传感器数据
            sensor_data = self.sensor_manager.get_latest()
            if sensor_data:
                self.mqtt.publish('sensor/data', sensor_data)

            # 检查超时
            self.controller.check_timeout()

            time.sleep(1)

    def cleanup(self):
        """清理资源"""
        self.sensor_manager.stop()
        self.tracker.stop()
        self.controller.stop()
        self.driver.cleanup()
        self.mqtt.disconnect()

if __name__ == '__main__':
    app = MainApp()

    def signal_handler(sig, frame):
        app.cleanup()
        sys.exit(0)

    signal.signal(signal.SIGINT, signal_handler)
    app.run()
